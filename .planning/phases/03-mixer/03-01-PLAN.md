let masterGain = null;

export function getMasterGain() {
  if (!masterGain) {
    const ctx = getContext();
    masterGain = ctx.createGain();
    masterGain.gain.value = 1.0;
    masterGain.connect(ctx.destination);
  }
  return masterGain;
}
